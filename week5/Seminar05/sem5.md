int removeDuplicates(int arr[], int size) {
    int newSize = size;  // –ù–æ–≤ —Ä–∞–∑–º–µ—Ä –Ω–∞ –º–∞—Å–∏–≤–∞

    for (int i = 0; i < newSize; i++) {

    // –ö–æ–ø–∏—Ä–∞–º–µ –≤—Å–∏—á–∫–∏ –µ–ª–µ–º–µ–Ω—Ç–∏ –æ—Ç –ø—ä—Ä–≤–æ—Ç–æ –º–Ω–æ–∂–µ—Å—Ç–≤–æ –≤ —Ä–µ–∑—É–ª—Ç–∞—Ç–Ω–∏—è –º–∞—Å–∏–≤

// –§—É–Ω–∫—Ü–∏—è –∑–∞ —Å–æ—Ä—Ç–∏—Ä–∞–Ω–µ –Ω–∞ –º–∞—Å–∏–≤ —Å Bubble Sort
void bubbleSort(int arr[], int size) {
    for (int i = 0; i < size - 1; i++) {
        // –ü—Ä–µ–º–∏–Ω–∞–≤–∞–º–µ –ø—Ä–µ–∑ –º–∞—Å–∏–≤–∞, –∫–∞—Ç–æ –Ω–∞–º–∞–ª—è–≤–∞–º–µ —Ä–∞–∑–º–µ—Ä–∞ –Ω–∞ –Ω–µ—Å–æ—Ä—Ç–∏—Ä–∞–Ω–∞—Ç–∞ —á–∞—Å—Ç
        for (int j = 0; j < size - 1 - i; j++) {
            // –ê–∫–æ –µ–ª–µ–º–µ–Ω—Ç—ä—Ç –µ –ø–æ-–≥–æ–ª—è–º –æ—Ç —Å–ª–µ–¥–≤–∞—â–∏—è, –≥–∏ —Ä–∞–∑–º–µ–Ω—è–º–µ
            if (arr[j] > arr[j + 1]) {
                swap(arr[j], arr[j + 1]);
            }
        }
    }
}

int main() {
    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int size = sizeof(arr) / sizeof(arr[0]);

    cout << "–ü—Ä–µ–¥–∏ —Å–æ—Ä—Ç–∏—Ä–∞–Ω–µ: ";
    for (int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;

    bubbleSort(arr, size);  

    cout << "–°–ª–µ–¥ —Å–æ—Ä—Ç–∏—Ä–∞–Ω–µ: ";
    for (int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;

    return 0;
}
